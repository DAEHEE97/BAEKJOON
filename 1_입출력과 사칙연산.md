```python
# map에 int와 input().split()을 넣으면 split의 결과를 모두 int로 변환해줍니다
# 실수로 변환할 때는 int대신 float를 넣습니다.

a, b = map(int, input().split())
print(a + b)
```

    100 3
    103



```python
# 입력받은 값을 콤마를 기준으로 분리

a, b = map(int, input().split(',')) 
print(a + b)


```

    100,3
    103
